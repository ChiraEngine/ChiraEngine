171  this->editor.SetLanguageDefinition(TextEditor::LanguageDefinition::C());
172  } else if (this->path.ends_with(".cpp") ||
173  this->path.ends_with(".h") ||

176  this->path.ends_with(".hpp") ||
177  this->path.ends_with(".ixx")) {
178  this->editor.SetLanguageDefinition(TextEditor::LanguageDefinition::CPlusPlus());
179  } else if (this->path.ends_with(".glsl") ||
180  this->path.ends_with(".vert") ||

182  this->path.ends_with(".frag") ||
183  this->path.ends_with(".fsh") ||
184  this->path.ends_with(".geom") ||

186  this->path.ends_with(".comp") ||
187  this->path.ends_with(".csh")) {

189  } else if (this->path.ends_with(".hlsl")) {
190  this->editor.SetLanguageDefinition(TextEditor::LanguageDefinition::HLSL());
191  } else if (this->path.ends_with(".lua")) {
192  this->editor.SetLanguageDefinition(TextEditor::LanguageDefinition::Lua());
193  } else {
194  this->editor.SetLanguageDefinition(TextEditor::LanguageDefinition{});
